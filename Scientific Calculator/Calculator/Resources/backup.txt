        private void funcal()
        {
            switch (op)
            {
                case "+":
                    textBox1.Text = Convert.ToString(Convert.ToDouble(value1) + Convert.ToDouble(textBox1.Text)); //adding values of textbox

                    break;
                case "-":
                    textBox1.Text = Convert.ToString(Convert.ToDouble(value1) - Convert.ToDouble(textBox1.Text));

                    break;
                case "*":
                    textBox1.Text = Convert.ToString(Convert.ToDouble(value1) * Convert.ToDouble(textBox1.Text));
                    break;
                case "/":
                    if (textBox1.Text == "0")
                    {
                        textBox1.Text = "infinity";
                    }
                    else
                    {
                        textBox1.Text = Convert.ToString(Convert.ToDouble(value1) / Convert.ToDouble(textBox1.Text));
                    }
                    break;
                case "^":
                    textBox1.Text = Convert.ToString(System.Math.Pow(Convert.ToDouble(value1), Convert.ToDouble(textBox1.Text)));
                    break;
                case "%": textBox1.Text = Convert.ToString(Convert.ToDouble(value1) % Convert.ToDouble(textBox1.Text));
                    break;
                case "P": double varn, var2, var3; //variable declaration 
                    varn = factorial(Convert.ToDouble(value1)); //calling factorial function 
                    var2 = factorial(Convert.ToDouble(value1) - Convert.ToDouble(textBox1.Text));
                    textBox1.Text = Convert.ToString(varn / var2); //storing or showing result of factorial variables
                    break;
                case "C":
                    varn = factorial(Convert.ToDouble(value1));
                    var2 = factorial(Convert.ToDouble(value1) - Convert.ToDouble(textBox1.Text));
                    var3 = factorial(Convert.ToDouble(textBox1.Text));
                    textBox1.Text = Convert.ToString(varn / (var3 * var2));
                    break;
                default:
                    break;
            }
        }

        private double factorial(double x)
        {
            double i = 1; //initialization values of i to 1
            for (double s = 1; s <= x; s++)
            {
                i = i * s;
            }
            return i;
        }
